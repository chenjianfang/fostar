├── config // 构建配置文件

├── scripts // 构建执行文件
│   └── build.js
│   └── start.js
├── src // 项目代码
│   ├── assets // 经过编译的静态文件

│   	└── constants // 常量

│   └── config // 项目
│   	└── menus // 菜单
│   	└── routes // 路由
│   	└── ... // 其他项目配置

│   └── layouts // 项目布局文件
│   	└── LayoutBase // 业务基本布局

│   └── pages // 页面
│   └── services // 页面
│   └── store // 状态管理
│   └── App.js
│   └── index.js // 入口文件
